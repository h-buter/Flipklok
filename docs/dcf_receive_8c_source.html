185bool checkBitStream()
186{
187 __no_operation();
188 // First bit needs to be 0, bit 20 needs to be 1, bit 59 needs to be 0
189 if (1 == bitArray[0] || 0 == bitArray[20] || 1 == bitArray[59])
190 {
191 __no_operation();
192 #ifdef UART_ENABLED
193 UART_SendString("RX bit fault\r\n");
194 #endif
195 return 1; //transmission fault,
196 }
197
198 __no_operation();
199 //Check parity minutes
200 unsigned int i;
201 unsigned int onesMinute = 0;
202 for(i = 21; i <= 27; i++) //minutes
203 {
204 if(1 == bitArray[i])
205 {
206 onesMinute++;
207 }
208 }
209
210 onesMinute = onesMinute + (int)bitArray[28]; // Add parity bit to even count, if the count is uneven the pariybit should make it even.
211
212 __no_operation();
213 if (onesMinute % 2) // if odd then transmission fault
214 {
215 __no_operation();
216 #ifdef UART_ENABLED
217 UART_SendString("RX fault no parity minute\r\n");
218 #endif
219 return 1; //transmission fault no even parity minute
220 }
221
222 __no_operation();
223 //Check parity hour
224 unsigned int j;
225 unsigned int onesHour = 0;
226 for(j = 29; j <= 34; j++) //hour
227 {
228 if(1 == bitArray[j])
229 {
230 onesHour++;
231 }
232 }
233
234 onesHour = onesHour + (int)bitArray[35]; // Add parity bit to even count, if the count is uneven the pariybit should make it even.
235
236 __no_operation();
237 if (onesHour % 2) // if odd then transmission fault
238 {
239
240 #ifdef UART_ENABLED
241 UART_SendString("RX fault no parity hour\r\n");
242 #endif
243 return 1; //transmission fault no even parity hour
244 }
245
246 __no_operation();
247 return 0; // No errors found
248 //other parity for date needs no checking because it is not used
